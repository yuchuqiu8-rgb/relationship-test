// ---------- START ----------

function startTest(){

  name = document.getElementById("nameInput").value || "TA";

  document.getElementById("start").classList.add("hidden");

  document.getElementById("quiz").classList.remove("hidden");

  showQuestion();

}

 

// ---------- SHOW ----------

function showQuestion(){

  let q = questions[current];

 

  document.getElementById("qIndex").innerText = current + 1;

  document.getElementById("question").innerText = q.q;

 

  // progress

  let percent = ((current) / questions.length) * 100;

  document.getElementById("progress").style.width = percent + "%";

 

  let html = "";

 

  q.options.forEach((opt)=>{

    html += `

      <button class="w-full p-3 border rounded-2xl hover:bg-gray-100 transition"

        onclick='selectOption(${JSON.stringify(opt.val)})'>

        ${opt.text}

      </button>

    `;

  });

 

  document.getElementById("options").innerHTML = html;

}

 

// ---------- SELECT ----------

function selectOption(val){

  for(let k in val){

    score[k] += val[k];

  }

 

  current++;

 

  if(current < questions.length){

    showQuestion();

  }else{

    showResult();

  }

}

 

// ---------- RESULT ----------
